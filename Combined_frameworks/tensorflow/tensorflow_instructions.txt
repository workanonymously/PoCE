1. Version can be changed by changing that on tensorflow_requirements.txt
2. After version change: create a dockerfile inside docker folder and save as ockerfile.tensorflow-version

FROM python:3.10-slim

WORKDIR /app

ENV PYTHONUNBUFFERED=1
ENV PYTHONPATH=/app
ENV IN_DOCKER=1
ENV TF_VERSION=version_number

COPY tensorflow/tensorflow_requirements.txt /tmp/tensorflow_requirements.txt
RUN pip install --no-cache-dir -r /tmp/tensorflow_requirements.txt

COPY . /app/Combined_frameworks

WORKDIR /app/Combined_frameworks

ENTRYPOINT ["python3", "main.py"]
3. After that build the image by running: docker build -f docker/dockerfile.tensorflow-2.17.0 -t tensorflow-tool:2.17.0 . "replace filename"
4. Update the path in tensorflow_tool.py
5. run as python3 main.py --frameworks --tensorflow
